    <script>
        // Funções de controle
        function startScreenCapture() {
            document.getElementById('screen-status').textContent = '✓ Captura iniciada';
        }
        function stopScreenCapture() {
            document.getElementById('screen-status').textContent = '✗ Captura parada';
        }
        function startAudio() {
            document.getElementById('audio-status').textContent = '✓ Áudio ativo';
        }
        function stopAudio() {
            document.getElementById('audio-status').textContent = '✗ Áudio parado';
        }
        function startVideo() {
            navigator.mediaDevices.getUserMedia({ video: true })
                .then(stream => {
                    document.getElementById('video').srcObject = stream;
                    document.getElementById('video-status').textContent = '✓ Câmera ativa';
                })
                .catch(err => {
                    document.getElementById('video-status').textContent = '✗ Erro: ' + err.message;
                });
        }
        function stopVideo() {
            const video = document.getElementById('video');
            if (video.srcObject) {
                video.srcObject.getTracks().forEach(track => track.stop());
                document.getElementById('video-status').textContent = '✗ Câmera parada';
            }
        }
    </script>
